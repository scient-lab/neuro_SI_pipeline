--extra-index-url https://download.pytorch.org/whl/cu121
# torch==2.5.1+cu121 (not 2.4.0) — vllm==0.7.3 below hard-requires torch 2.5.1
# and uv cannot resolve the older pin against it. Matches the graphmert
# venv's torch version so they share the cu121 wheels in HF/uv cache.
# History: was 2.4.0+cu121 before vllm pin was added explicitly today;
# the transitive vllm install (some older version) was compatible with
# 2.4.0, but vllm 0.7.3 is what we need per [Stage 3 RL 12GB gotchas]
# (LoRA-on-RL + TRL API alignment) so the bump is forced.
torch==2.5.1+cu121
# setuptools: uv venvs don't ship setuptools by default. Required AT
# RUNTIME for torch.utils.cpp_extension (deepspeed imports it during
# model init for CUDA introspection) AND at BUILD time for any package
# using legacy setup.py-style builds (e.g. flash-attn). Surfaced as bug
# #10 on 2026-06-24 when trainer.py hit ModuleNotFoundError mid-init.
# Pinning the major version since 70+ removed the wheel/bdist_wheel
# command that older packaging tools (incl. uv) still call.
setuptools>=70,<82
# transformers PINNED to match 2_graphmert/requirements.txt's known-good
# range. Unpinned transformers (the prior state) let uv resolve >=4.58,
# which refactored Qwen2Tokenizer.all_special_tokens_extended and added
# strict-validating MistralCommonBackend — both incompatible with the
# vllm==0.7.3 tokenizer wrapper. Surfaced as a 24h debug chain on
# 2026-06-24 (validate_qa step). See _tokenizer_compat.py for the
# monkey-patches that were the prior workaround; with this pin those
# can be removed eventually.
transformers>=4.55,<4.58
# trl version constraints (locally verified 2026-06-24):
#   - GRPO classes were added in trl 0.14.0 (NOT 0.12.0 as initially
#     assumed); versions <0.14 have no GRPOConfig/GRPOTrainer at all.
#   - trl 0.14.0 through 0.17.x import cleanly with torch==2.5.1+cu121
#     (no FSDPModule symbol error).
#   - trl 0.18.0+ adds the FSDPModule import path which needs torch>=2.6;
#     incompatible with our vllm 0.7.3 pin that holds torch at 2.5.1.
# Locally tested: uv venv + torch==2.5.1+cu121 + trl==0.14.0/0.15.0/
# 0.16.0/0.17.0 all import GRPOConfig/GRPOTrainer successfully.
# Memory entry [Stage 3 RL 12GB gotchas] flagged TRL API drift as a
# known fork-patch class — this pin formalizes the safe window.
trl>=0.14,<0.18
peft
deepspeed
# flash-attn powers attn_implementation=flash_attention_2 in trainer.py
# (configs/default.yaml::sft.attn_implementation). The paper used FA2 on
# Princeton's Della HPC.
#
# We install a PRE-BUILT WHEEL, not the sdist. flash-attn is compiled CUDA C++,
# and its source build has chained missing-dep errors in uv's isolated build
# env (torch → setuptools → packaging → …) that never converge. The wheel skips
# the compile entirely — pip just unpacks a binary someone already built.
#
# The wheel tags MUST match THIS venv exactly, or it won't import:
#   cu12 · torch2.5 · cxx11abiFALSE · cp310 · linux_x86_64
#   (python 3.10 per setup.sh, torch==2.5.1+cu121 pinned above, ABI=False, Ada+ GPU).
# If you bump torch/python/CUDA, pick the matching asset from
# https://github.com/Dao-AILab/flash-attention/releases and update this URL.
# With this present, attn_implementation=flash_attention_2 works and NO profile
# needs the sdpa override. (A non-Ada/Hopper GPU without sm_80+ still can't use
# FA2 — fall back to sdpa there.)
flash_attn @ https://github.com/Dao-AILab/flash-attention/releases/download/v2.7.4.post1/flash_attn-2.7.4.post1+cu12torch2.5cxx11abiFALSE-cp310-cp310-linux_x86_64.whl
# accelerate pin propagated from 2_graphmert/requirements.txt (the venv where
# this pin was originally tested). transformers Trainer hard-requires
# accelerate; unpinned resolution can pick a version with API drift vs
# our transformers/trl combo. <2 is the future-proof ceiling.
accelerate>=0.34,<2
# datasets pin propagated from 2_graphmert/requirements.txt with the
# original explanatory comment:

# was already in graphmert's file, we just never propagated the pin.
datasets>=2.20,<4
